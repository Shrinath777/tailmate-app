import { Stack, useRouter, useSegments } from 'expo-router';
import { AuthProvider, useAuth } from '../hooks/useAuth';
import { useEffect } from 'react';
import { View, ActivityIndicator } from 'react-native';
import { Colors } from '../constants/Colors';

// Auth redirect component
function AuthRedirect() {
  const { user, isCheckingAuth } = useAuth();
  const segments = useSegments();
  const router = useRouter();

  useEffect(() => {
    if (isCheckingAuth) return;

    // Check the current route segment
    const currentRoute = segments[0];
    
    if (user) {
      // User is authenticated - redirect from auth to tabs
      if (currentRoute === '(auth)') {
        router.replace('/(tabs)');
      }
    } else {
      // User is not authenticated - redirect from tabs to auth
      if (currentRoute === '(tabs)') {
        router.replace('/(auth)/login');
      }
    }
  }, [user, isCheckingAuth, segments]);

  if (isCheckingAuth) {
    return (
      <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center' }}>
        <ActivityIndicator size="large" color={Colors.primary} />
      </View>
    );
  }

  return null;
}

export default function RootLayout() {
  return (
    <AuthProvider>
      <AuthRedirect />
      <Stack screenOptions={{ headerShown: false }}>
        {/* Don't define route groups here - they are automatically handled */}
        {/* Just define individual screens if needed, or leave empty */}
      </Stack>
    </AuthProvider>
  );
} working code for the authenticate not auto login
